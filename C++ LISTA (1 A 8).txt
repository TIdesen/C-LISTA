#include <iostream>
#include <vector>

using namespace std;

int main() {
    const int NUM_MESAS = 30, LUGARES_POR_MESA = 5;
    vector<int> reservas(NUM_MESAS, 0);
    int total_lugares_ocupados = 0, codigo_mesa;

    while (total_lugares_ocupados < NUM_MESAS * LUGARES_POR_MESA) {
        cout << "Código da mesa (100 a 129) ou 0 para sair: ";
        cin >> codigo_mesa;

        if (codigo_mesa == 0) break;

        if (codigo_mesa < 100 || codigo_mesa > 129) {
            cout << "Código inválido!" << endl;
            continue;
        }

        int indice_mesa = codigo_mesa - 100;
        int lugares_desejados;

        cout << "Lugares desejados (1 a 5): ";
        cin >> lugares_desejados;

        if (lugares_desejados < 1 || lugares_desejados > LUGARES_POR_MESA ||
            reservas[indice_mesa] + lugares_desejados > LUGARES_POR_MESA) {
            cout << "Reserva inválida!" << endl;
            continue;
        }

        reservas[indice_mesa] += lugares_desejados;
        total_lugares_ocupados += lugares_desejados;

        cout << "Reserva na mesa " << codigo_mesa << " realizada!" << endl;
    }

    cout << "Total de lugares ocupados: " << total_lugares_ocupados << "/" << (NUM_MESAS * LUGARES_POR_MESA) << endl;

    return 0;
}
